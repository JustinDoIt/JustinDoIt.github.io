import numpy as np

##定义sigmoid函数
def sigmoid(x):
    return 1/(1+np.exp(-x))

##初始化参数
def initialization_parameters(dim):
    w = np.zeros((dim,1))
    b= 0.0
    return w, b

##前向传播
def forward_propagate(w,b, X,Y):
    m = X.shape[1]

    A = np.dot(w.T, X) + b
    y_hat = sigmoid(A)
    cost = -1/m * np.sum(Y*np.log(y_hat) + (1-Y)*np.log(1-y_hat))

    ##计算梯度
    dw = np.dot(X, (y_hat-Y).T)/m
    db = np.sum(y_hat-Y)/m

    grad = { 'dw':dw,
             'db':db
    }
    return grad, cost

##后向传播,调整参数
def back_propagate(w,b,grad,alpha):
    dw = grad['dw']
    db = grad['db']
    w = w - alpha*dw
    b = b - alpha*db

    return w,b
